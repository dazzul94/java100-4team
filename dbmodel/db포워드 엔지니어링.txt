-- 후기
CREATE TABLE PostScript (
	rent_no  INTEGER      NOT NULL COMMENT '대여번호', -- 대여번호
	cnt      VARCHAR(255) NULL     COMMENT '내용', -- 내용
	cre_date DATE         NULL     COMMENT '작성일' -- 작성일
)
COMMENT '후기';

-- 후기
ALTER TABLE PostScript
	ADD CONSTRAINT PK_PostScript -- 후기 기본키
		PRIMARY KEY (
			rent_no -- 대여번호
		);

-- 후기 인덱스
CREATE INDEX IX_PostScript
	ON PostScript( -- 후기
		cre_date ASC -- 작성일
	);

-- 물품

	it_no     INTEGER     NOT NULL COMMENT '물품번호', -- 물품번호

	name      VARCHAR(50) NOT NULL COMMENT '물품명', -- 물품명
	start     DATE        NOT NULL COMMENT '대여시작일', -- 대여시작일
	end       DATE        NOT NULL COMMENT '대여종료일', -- 대여종료일
	price     INTEGER     NOT NULL COMMENT '대여 기본 가격', -- 대여 기본 가격
	price_day INTEGER     NOT NULL COMMENT '대여 가격', -- 대여 가격
	status    BOOLEAN     NOT NULL COMMENT '물품 상태' -- 물품 상태

COMMENT '물품';

-- 물품
ALTER TABLE Item
	ADD CONSTRAINT PK_Item -- 물품 기본키
		PRIMARY KEY (
			it_no -- 물품번호
		);

	ON Item( -- 물품

	);

ALTER TABLE Item

ALTER TABLE Account
	ADD CONSTRAINT PK_Account -- 계정 기본키

	);

-- 사용자
